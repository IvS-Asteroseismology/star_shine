STAR SHINE Satellite Time-series Analysis Routine using Sinusoids and Harmonics through Iterative Non-linear Extraction

This Python module contains classes for handling the models for the time series. Includes a piece-wise linear model, sum of sinusoids, and harmonics.

  15@nb.njit(cache=True, parallel=True)
  16def linear_curve(time, const, slope, i_chunks, t_shift=True):
  17    """Returns a piece-wise linear curve for the given time points
  18    with slopes and y-intercepts.
  19
  20    Parameters
  21    ----------
  22    time: numpy.ndarray[Any, dtype[float]]
  23        Timestamps of the time series
  24    const: numpy.ndarray[Any, dtype[float]]
  25        The y-intercepts of a piece-wise linear curve
  26    slope: numpy.ndarray[Any, dtype[float]]
  27        The slopes of a piece-wise linear curve
  28    i_chunks: numpy.ndarray[Any, dtype[int]]
  29        Pair(s) of indices indicating time chunks within the light curve, separately handled in cases like
  30        the piecewise-linear curve. If only a single curve is wanted, set to np.array([[0, len(time)]]).
  31    t_shift: bool
  32        Mean center the time axis
  33
  34    Returns
  35    -------
  36    numpy.ndarray[Any, dtype[float]]
  37        The model time series of a (set of) straight line(s)
  38
  39    Notes
  40    -----
  41    Assumes the constants and slopes are determined with respect to the sector mean time as zero point.
  42    """
  43    curve = np.zeros(len(time))
  44    for i in nb.prange(len(const)):
  45        s = i_chunks[i]
  46        if t_shift:
  47            t_sector_mean = np.mean(time[s[0]:s[1]])
  48        else:
  49            t_sector_mean = 0
  50
  51        curve[s[0]:s[1]] = const[i] + slope[i] * (time[s[0]:s[1]] - t_sector_mean)
  52
  53    return curve

  56@nb.njit(cache=True, parallel=True)
  57def linear_pars(time, flux, i_chunks):
  58    """Calculate the slopes and y-intercepts of a linear trend with the MLE.
  59
  60    Parameters
  61    ----------
  62    time: numpy.ndarray[Any, dtype[float]]
  63        Timestamps of the time series
  64    flux: numpy.ndarray[Any, dtype[float]]
  65        Measurement values of the time series
  66    i_chunks: numpy.ndarray[Any, dtype[int]]
  67        Pair(s) of indices indicating time chunks within the light curve, separately handled in cases like
  68        the piecewise-linear curve. If only a single curve is wanted, set to np.array([[0, len(time)]]).
  69
  70    Returns
  71    -------
  72    tuple
  73        A tuple containing the following elements:
  74        y_inter: numpy.ndarray[Any, dtype[float]]
  75            The y-intercepts of a piece-wise linear curve
  76        slope: numpy.ndarray[Any, dtype[float]]
  77            The slopes of a piece-wise linear curve
  78
  79    Notes
  80    -----
  81    Source: https://towardsdatascience.com/linear-regression-91eeae7d6a2e
  82    Determines the constants and slopes with respect to the sector mean time as zero point to avoid correlations.
  83    """
  84    y_inter = np.zeros(len(i_chunks))
  85    slope = np.zeros(len(i_chunks))
  86
  87    for i in nb.prange(len(i_chunks)):
  88        s = i_chunks[i]
  89
  90        # mean and mean subtracted quantities
  91        x_m = np.mean(time[s[0]:s[1]])
  92        x_ms = (time[s[0]:s[1]] - x_m)
  93        y_m = np.mean(flux[s[0]:s[1]])
  94        y_ms = (flux[s[0]:s[1]] - y_m)
  95
  96        # sums
  97        s_xx = np.sum(x_ms ** 2)
  98        s_xy = np.sum(x_ms * y_ms)
  99
 100        # parameters
 101        slope[i] = s_xy / s_xx
 102        # y_inter[i] = y_m - slope[i] * x_m  # original non-mean-centered formula
 103        y_inter[i] = y_m  # mean-centered value
 104
 105    return y_inter, slope

 109def sum_sines_st(time, f_n, a_n, ph_n, t_shift=True):
 110    """A sum of sinusoids at times t, given the frequencies, amplitudes and phases.
 111
 112    Single threaded version. Better for one to a few sinusoids.
 113
 114    Parameters
 115    ----------
 116    time: numpy.ndarray[Any, dtype[float]]
 117        Timestamps of the time series
 118    f_n: numpy.ndarray[Any, dtype[float]]
 119        The frequencies of a number of sinusoids
 120    a_n: numpy.ndarray[Any, dtype[float]]
 121        The amplitudes of a number of sinusoids
 122    ph_n: numpy.ndarray[Any, dtype[float]]
 123        The phases of a number of sinusoids
 124    t_shift: bool
 125        Mean center the time axis
 126
 127    Returns
 128    -------
 129    numpy.ndarray[Any, dtype[float]]
 130        Model time series of a sum of sinusoids. Varies around 0.
 131
 132    Notes
 133    -----
 134    Assumes the phases are determined with respect to the mean time as zero point by default.
 135    """
 136    if t_shift:
 137        mean_t = np.mean(time)
 138    else:
 139        mean_t = 0
 140
 141    model_sines = np.zeros(len(time))
 142    for i in range(len(f_n)):
 143        model_sines += a_n[i] * np.sin((2 * np.pi * f_n[i] * (time - mean_t)) + ph_n[i])
 144
 145    return model_sines

 186@nb.njit(cache=True, parallel=True)
 187def sum_sines_deriv(time, f_n, a_n, ph_n, deriv=1, t_shift=True):
 188    """The time derivative of a sum of sinusoids at times t.
 189
 190    Parameters
 191    ----------
 192    time: numpy.ndarray[Any, dtype[float]]
 193        Timestamps of the time series
 194    f_n: list[float], numpy.ndarray[Any, dtype[float]]
 195        The frequencies of a number of sinusoids
 196    a_n: list[float], numpy.ndarray[Any, dtype[float]]
 197        The amplitudes of a number of sinusoids
 198    ph_n: list[float], numpy.ndarray[Any, dtype[float]]
 199        The phases of a number of sinusoids
 200    deriv: int
 201        Number of time derivatives taken (>= 1)
 202    t_shift: bool
 203        Mean center the time axis
 204
 205    Returns
 206    -------
 207    numpy.ndarray[Any, dtype[float]]
 208        Model time series of a sum of sinusoid derivatives. Varies around 0.
 209
 210    Notes
 211    -----
 212    Assumes the phases are determined with respect to the mean time as zero point by default.
 213    """
 214    if t_shift:
 215        mean_t = np.mean(time)
 216    else:
 217        mean_t = 0
 218
 219    model_sines = np.zeros(len(time))
 220    mod_2 = deriv % 2
 221    mod_4 = deriv % 4
 222    ph_cos = (np.pi / 2) * mod_2  # alternate between cosine and sine
 223    sign = (-1) ** ((mod_4 - mod_2) // 2)  # (1, -1, -1, 1, 1, -1, -1... for deriv=1, 2, 3...)
 224
 225    for i in nb.prange(len(f_n)):
 226        for j in range(len(time)):
 227            model_sines[j] += (sign * (2 * np.pi * f_n[i]) ** deriv * a_n[i] *
 228                               np.sin((2 * np.pi * f_n[i] * (time[j] - mean_t)) + ph_n[i] + ph_cos))
 229
 230    return model_sines
