 20class Result:
 21    """A class to handle analysis results.
 22
 23    Attributes
 24    ----------
 25    target_id: str
 26        User defined identification number or name for the target under investigation.
 27    data_id: str
 28        User defined identification name for the dataset used.
 29    n_param: int
 30        Number of free parameters in the model.
 31    bic: float
 32        Bayesian Information Criterion of the residuals.
 33    noise_level: float
 34        The noise level (standard deviation of the residuals).
 35    const: numpy.ndarray[Any, dtype[float]]
 36        The y-intercepts of a piece-wise linear curve.
 37    const_err: numpy.ndarray[Any, dtype[float]]
 38        Uncertainty in the constant for each time chunk.
 39    const_hdi: numpy.ndarray[Any, dtype[float]]
 40        HDI bounds for the constant for each time chunk.
 41    slope: numpy.ndarray[Any, dtype[float]]
 42        The slopes of a piece-wise linear curve.
 43    slope_err: numpy.ndarray[Any, dtype[float]]
 44        Uncertainty in the slope for each time chunk.
 45    slope_hdi: numpy.ndarray[Any, dtype[float]]
 46        HDI bounds for the slope for each time chunk.
 47    f_n: numpy.ndarray[Any, dtype[float]]
 48        The frequencies of a number of sinusoids.
 49    f_n_err: numpy.ndarray[Any, dtype[float]]
 50        Uncertainty in the frequency for each sinusoid.
 51    f_n_hdi: numpy.ndarray[Any, dtype[float]]
 52        HDI bounds for the frequency for each sinusoid.
 53    a_n: numpy.ndarray[Any, dtype[float]]
 54        The amplitudes of a number of sinusoids.
 55    a_n_err: numpy.ndarray[Any, dtype[float]]
 56        Uncertainty in the amplitude for each sinusoid (these are identical).
 57    a_n_hdi: numpy.ndarray[Any, dtype[float]]
 58        HDI bounds for the amplitude for each sinusoid.
 59    ph_n: numpy.ndarray[Any, dtype[float]]
 60        The phases of a number of sinusoids.
 61    ph_n_err: numpy.ndarray[Any, dtype[float]]
 62        Uncertainty in the phase for each sinusoid.
 63    ph_n_hdi: numpy.ndarray[Any, dtype[float]]
 64        HDI bounds for the phase for each sinusoid.
 65    passing_sigma: numpy.ndarray[Any, dtype[bool]]
 66        Sinusoids that passed the sigma check.
 67    passing_snr: numpy.ndarray[Any, dtype[bool]]
 68        Sinusoids that passed the signal-to-noise check.
 69    h_base: numpy.ndarray[Any, dtype[int]]
 70        Indices of the base frequencies in _f_n for each of _f_n. -1 indicates a free sinusoid.
 71    h_mult: numpy.ndarray[Any, dtype[int]]
 72        Harmonic multiplier of the base frequency for each of _f_n. 0 indicates a free sinusoid.
 73    f_h_err: numpy.ndarray[Any, dtype[float]]
 74        Uncertainty in the frequency for each harmonic sinusoid.
 75    passing_harmonic: numpy.ndarray[Any, dtype[bool]]
 76        Sinusoids that passed the harmonic check.
 77    """

 79    def __init__(self):
 80        """Initialises the Result object."""
 81        # descriptive
 82        self.target_id = ''
 83        self.data_id = ''
 84
 85        # summary statistics
 86        self.n_param = -1
 87        self.bic = -1.
 88        self.noise_level = -1.
 89
 90        # attribute lists per model
 91        self.linear_property_list = ['const', 'const_err', 'const_hdi', 'slope', 'slope_err', 'slope_hdi']
 92        self.sinusoid_property_list = ['f_n', 'f_n_err', 'f_n_hdi', 'a_n', 'a_n_err', 'a_n_hdi',
 93                                       'ph_n', 'ph_n_err', 'ph_n_hdi', 'passing_sigma', 'passing_snr',
 94                                       'h_base', 'h_mult', 'f_h_err', 'passing_harmonic']
 95
 96        # linear model parameters
 97        # y-intercepts
 98        self.const = np.zeros((0,))
 99        self.const_err = np.zeros((0,))
100        self.const_hdi = np.zeros((0, 2))
101        # slopes
102        self.slope = np.zeros((0,))
103        self.slope_err = np.zeros((0,))
104        self.slope_hdi = np.zeros((0, 2))
105
106        # sinusoid model parameters
107        # frequencies
108        self.f_n = np.zeros((0,))
109        self.f_n_err = np.zeros((0,))
110        self.f_n_hdi = np.zeros((0, 2))
111        # amplitudes
112        self.a_n = np.zeros((0,))
113        self.a_n_err = np.zeros((0,))
114        self.a_n_hdi = np.zeros((0, 2))
115        # phases
116        self.ph_n = np.zeros((0,))
117        self.ph_n_err = np.zeros((0,))
118        self.ph_n_hdi = np.zeros((0, 2))
119        # passing criteria
120        self.passing_sigma = np.zeros((0,), dtype=bool)
121        self.passing_snr = np.zeros((0,), dtype=bool)
122
123        # harmonic model
124        self.h_base = np.zeros((0,), dtype=int)
125        self.h_mult = np.zeros((0,), dtype=int)
126        self.f_h_err = np.zeros((0,))
127        # passing criteria
128        self.passing_harmonic = np.zeros((0,), dtype=bool)
129
130        return

176    def from_time_series_model(self, ts_model, target_id=None, data_id=None):
177        """Fill in the Result attributes with results from a TimeSeriesModel.
178
179        Parameters
180        ----------
181        ts_model: tms.TimeSeriesModel
182            Instance of TimeSeriesModel containing the time series and model parameters.
183        target_id: str
184            User defined identification number or name for the target under investigation.
185        data_id: str
186            User defined identification name for the dataset used.
187        """
188        # descriptive
189        self.target_id = target_id
190        self.data_id = data_id
191
192        # summary statistics
193        self.n_param = ts_model.n_param
194        self.bic = ts_model.bic()
195        self.noise_level = ut.std_unb(ts_model.residual(), ts_model.n_time - ts_model.n_param)
196
197        # linear model parameters
198        for key in self.linear_property_list:
199            if '_hdi' not in key:  # avoid hdi for now
200                setattr(self, key, getattr(ts_model.linear, key))
201
202        # sinusoid model parameters
203        ts_model.remove_excluded()  # clean up before transfer
204        for key in self.sinusoid_property_list:
205            if '_hdi' not in key:  # avoid hdi for now
206                setattr(self, key, getattr(ts_model.sinusoid, key))
207
208        return None

211    def load(cls, file_name, h5py_file_kwargs=None, logger=None):
212        """Load a result file in hdf5 format.
213
214        Parameters
215        ----------
216        file_name: str
217            File name to load the results from
218        h5py_file_kwargs: dict, optional
219            Keyword arguments for opening the h5py file.
220            Example: {'locking': False}, for a drive that does not support locking.
221        logger: logging.Logger, optional
222            Instance of the logging library.
223
224        Returns
225        -------
226        Result
227            Instance of the Result class with the loaded results.
228        """
229        # guard for not existing file
230        if not os.path.isfile(file_name):
231            instance = cls()
232            return instance
233
234        # add everything to a dict
235        result_dict = io.load_result_hdf5(file_name, h5py_file_kwargs=h5py_file_kwargs)
236
237        # initiate the Results instance and fill in the results
238        instance = cls()
239        instance.from_dict(**result_dict)
240
241        if logger is not None:
242            logger.info(f"Loaded result file with target identifier: {result_dict['target_id']}, "
243                        f"created on {result_dict['date_time']}. Data identifier: {result_dict['data_id']}.")
244
245        return instance

247    def to_time_series_model(self, ts_model):
248        """Copy the parameters into an existing TimeSeriesModel instance.
249
250        Parameters
251        ----------
252        ts_model: tms.TimeSeriesModel
253            Instance of TimeSeriesModel containing the time series and model parameters.
254
255        Returns
256        -------
257        tms.TimeSeriesModel
258            Instance of TimeSeriesModel containing the time series and model parameters.
259        """
260        # linear model
261        ts_model.set_linear_model(self.const, self.slope)
262
263        # set the errors
264        ts_model.linear._const_err = self.const_err
265        ts_model.linear._slope_err = self.slope_err
266
267        # update numbers
268        ts_model.linear.update_n()
269
270        # sinusoid model
271        ts_model.set_sinusoids(self.f_n, self.a_n, self.ph_n, self.h_base, self.h_mult)
272
273        # set the errors
274        ts_model.sinusoid._f_n_err = self.f_n_err
275        ts_model.sinusoid._a_n_err = self.a_n_err
276        ts_model.sinusoid._ph_n_err = self.ph_n_err
277        ts_model.sinusoid._f_h_err = self.f_h_err
278
279        # set the criterion passing masks
280        ts_model.sinusoid._passing_sigma = self.passing_sigma
281        ts_model.sinusoid._passing_snr = self.passing_snr
282        ts_model.sinusoid._passing_harmonic = self.passing_harmonic
283
284        # update numbers
285        ts_model.sinusoid.update_n()
286
287        return ts_model
